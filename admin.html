🗄️ Créer la table Supabase
Copiez ce SQL et exécutez-le dans Supabase → SQL Editor pour créer la table actualites.
-- Table actualites pour SecuPRO Actu Sécu CREATE TABLE IF NOT EXISTS public.actualites ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, categorie text NOT NULL CHECK (categorie IN ('cnaps','conv','emploi')), titre text NOT NULL, resume text NOT NULL, contenu text NOT NULL, source text NOT NULL DEFAULT 'SecuPRO', date_publication timestamp with time zone DEFAULT now(), statut text NOT NULL DEFAULT 'brouillon' CHECK (statut IN ('brouillon','publié')), created_at timestamp with time zone DEFAULT now() ); -- Activer RLS (Row Level Security) ALTER TABLE public.actualites ENABLE ROW LEVEL SECURITY; -- Politique : lecture publique pour les articles publiés CREATE POLICY "Lecture publique articles publiés" ON public.actualites FOR SELECT USING (statut = 'publié'); -- Index pour les performances CREATE INDEX IF NOT EXISTS idx_actualites_statut ON public.actualites(statut); CREATE INDEX IF NOT EXISTS idx_actualites_categorie ON public.actualites(categorie); CREATE INDEX IF NOT EXISTS idx_actualites_date ON public.actualites(date_publication DESC);
⚠️ Note : Ce dashboard utilise les variables Supabase définies dans app.html. Assurez-vous de configurer les mêmes clés SUPABASE_URL et SUPABASE_KEY ci-dessous avant de gérer les articles.
